class Pair
{
    int first;
    int second;
}

Pair p := new Pair;
Pair q := new Pair;

void swap(Pair a)
{
    int tmp := a.first;
    a.first := a.second;
    a.second := tmp;
}

void main()
{
    p.first := 10;
    p.second := 20;
    q.first := 30;
    q.second := 40;
    swap(p);
    PrintInt(p.first);
    PrintInt(p.second);
    PrintInt(q.first);
    PrintInt(q.second);
}
