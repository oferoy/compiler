class Engine
{
    int power := 100;
    int getPower() { return power; }
}

class Car
{
    Engine engine;
    
    void setEngine(Engine e)
    {
        engine := e;
    }
    
    int getCarPower()
    {
        return engine.getPower();
    }
}

void main()
{
    Engine e1 := new Engine;
    e1.power := 250;
    
    Car c1 := new Car;
    c1.setEngine(e1);
    
    PrintInt(c1.getCarPower());
}
