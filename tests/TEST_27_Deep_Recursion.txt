int ackermann(int m, int n)
{
    if (m = 0)
    {
        return n + 1;
    }
    if (n = 0)
    {
        return ackermann(m - 1, 1);
    }
    return ackermann(m - 1, ackermann(m, n - 1));
}

void main()
{
    PrintInt(ackermann(0, 0));
    PrintInt(ackermann(1, 2));
    PrintInt(ackermann(2, 2));
    PrintInt(ackermann(3, 2));
}
