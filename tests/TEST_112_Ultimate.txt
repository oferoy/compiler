class Animal {
    string name := "unknown";
    int score(){ return 0; }
    string speak(){ return "silent"; }
}

class Cat extends Animal {
    int lives := 9;
    int score(){ return lives * 1000; }
    string speak(){ return "meow"; }
}

class Dog extends Animal {
    int tricks := 0;
    int score(){ return tricks * 500 + 100; }
    string speak(){ return "woof"; }
}

class Bird extends Animal {
    int wingspan := 1;
    int score(){ return wingspan * wingspan; }
    string speak(){ return "tweet"; }
}

array AnimalArr = Animal[];
array IntArr = int[];

class ScoreNode {
    int val;
    string who;
    ScoreNode next := nil;
}

int gcount := 0;

int bump()
{
    gcount := gcount + 1;
    return gcount;
}

ScoreNode insertSorted(ScoreNode head, int val, string who)
{
    ScoreNode n := new ScoreNode;
    n.val := val;
    n.who := who;
    if (head = nil) { return n; }
    if (val < head.val)
    {
        n.next := head;
        return n;
    }
    ScoreNode cur := head;
    while (cur.next = nil = 0)
    {
        if (val < cur.next.val)
        {
            n.next := cur.next;
            cur.next := n;
            return head;
        }
        cur := cur.next;
    }
    cur.next := n;
    return head;
}

int sumScores(ScoreNode n)
{
    if (n = nil) { return 0; }
    return n.val + sumScores(n.next);
}
