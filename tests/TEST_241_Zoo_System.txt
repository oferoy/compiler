array IntArr = int[];

class Animal
{
    string species := "unknown";
    int age := 0;

    string getSpecies() { return species; }
    int getAge() { return age; }
    int canBreed() { return age > 2; }
}

class Mammal extends Animal
{
    int legs := 4;

    int canBreed()
    {
        if (age > 1)
        {
            return 1;
        }
        return 0;
    }
}

class Reptile extends Animal
{
    int cold := 1;

    int canBreed()
    {
        if (age > 3)
        {
            return 1;
        }
        return 0;
    }
}

array AnimalArr = Animal[];

int countBreeders(AnimalArr zoo, int n)
{
    int count := 0;
    int i := 0;
    while (i < n)
    {
        if (zoo[i].canBreed())
        {
            count := count + 1;
        }
        i := i + 1;
    }
    return count;
}

void main()
{
    AnimalArr zoo := new Animal[4];
    Mammal lion := new Mammal;
    lion.species := "lion";
    lion.age := 5;
    Mammal puppy := new Mammal;
    puppy.species := "dog";
    puppy.age := 1;
    Reptile croc := new Reptile;
    croc.species := "croc";
    croc.age := 10;
    Reptile gecko := new Reptile;
    gecko.species := "gecko";
    gecko.age := 2;
    zoo[0] := lion;
    zoo[1] := puppy;
    zoo[2] := croc;
    zoo[3] := gecko;
    int i := 0;
    while (i < 4)
    {
        PrintString(zoo[i].getSpecies());
        PrintInt(zoo[i].getAge());
        PrintInt(zoo[i].canBreed());
        i := i + 1;
    }
    PrintInt(countBreeders(zoo, 4));
}
