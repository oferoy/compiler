array IntArr = int[];

void main()
{
    int n := 20;
    IntArr fib := new int[20];
    fib[0] := 0;
    fib[1] := 1;
    int i := 2;
    while (i < n)
    {
        fib[i] := fib[i - 1] + fib[i - 2];
        i := i + 1;
    }
    i := 0;
    while (i < n)
    {
        PrintInt(fib[i]);
        i := i + 1;
    }
}
