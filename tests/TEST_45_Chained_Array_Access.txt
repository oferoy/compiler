array IntArray = int[];

void main()
{
    IntArray indices := new int[5];
    IntArray values := new int[10];
    
    indices[0] := 2;
    indices[1] := 5;
    indices[2] := 8;
    indices[3] := 1;
    indices[4] := 9;
    
    int i := 0;
    while (i < 10)
    {
        values[i] := i * 10;
        i := i + 1;
    }
    
    PrintInt(values[indices[0]]);
    PrintInt(values[indices[1]]);
    PrintInt(values[indices[2]]);
}
