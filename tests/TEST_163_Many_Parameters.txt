int add(int a, int b, int c, int d, int e, int f, int g, int h)
{
    return a + b + c + d + e + f + g + h;
}

int weighted(int a, int b, int c, int d, int wa, int wb, int wc, int wd)
{
    return a * wa + b * wb + c * wc + d * wd;
}

void main()
{
    PrintInt(add(1, 2, 3, 4, 5, 6, 7, 8));
    PrintInt(add(100, 200, 300, 400, 500, 600, 700, 800));
    PrintInt(weighted(10, 20, 30, 40, 1, 2, 3, 4));
}
