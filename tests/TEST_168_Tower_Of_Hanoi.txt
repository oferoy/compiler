int moveCount := 0;

void hanoi(int n, int from, int to, int aux)
{
    if (n = 0)
    {
        return;
    }
    hanoi(n - 1, from, aux, to);
    moveCount := moveCount + 1;
    hanoi(n - 1, aux, to, from);
}

void main()
{
    hanoi(1, 1, 3, 2);
    PrintInt(moveCount);
    moveCount := 0;
    hanoi(3, 1, 3, 2);
    PrintInt(moveCount);
    moveCount := 0;
    hanoi(5, 1, 3, 2);
    PrintInt(moveCount);
    moveCount := 0;
    hanoi(10, 1, 3, 2);
    PrintInt(moveCount);
}
