int add(int a, int b, int c, int d)
{
    return a + b + c + d;
}

int mul(int a, int b)
{
    return a * b;
}

void main()
{
    int r := add(mul(2, 3), mul(4, 5), mul(6, 7), mul(8, 9));
    PrintInt(r);
    int s := add(add(1, 2, 3, 4), add(5, 6, 7, 8), add(9, 10, 11, 12), add(13, 14, 15, 16));
    PrintInt(s);
}
