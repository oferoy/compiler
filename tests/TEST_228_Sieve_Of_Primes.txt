array IntArr = int[];

void main()
{
    int n := 50;
    IntArr sieve := new int[50];
    int i := 0;
    while (i < n)
    {
        sieve[i] := 1;
        i := i + 1;
    }
    sieve[0] := 0;
    sieve[1] := 0;
    i := 2;
    while (i * i < n)
    {
        if (sieve[i])
        {
            int j := i * i;
            while (j < n)
            {
                sieve[j] := 0;
                j := j + i;
            }
        }
        i := i + 1;
    }
    i := 0;
    while (i < n)
    {
        if (sieve[i])
        {
            PrintInt(i);
        }
        i := i + 1;
    }
}
