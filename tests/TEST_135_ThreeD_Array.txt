array IntArr = int[];
array Matrix = IntArr[];
array Cube = Matrix[];

void main()
{
    Cube c := new Matrix[2];
    c[0] := new IntArr[2];
    c[1] := new IntArr[2];
    c[0][0] := new int[2];
    c[0][1] := new int[2];
    c[1][0] := new int[2];
    c[1][1] := new int[2];
    c[0][0][0] := 1;
    c[0][0][1] := 2;
    c[0][1][0] := 3;
    c[0][1][1] := 4;
    c[1][0][0] := 5;
    c[1][0][1] := 6;
    c[1][1][0] := 7;
    c[1][1][1] := 8;
    int sum := 0;
    int i := 0;
    while (i < 2)
    {
        int j := 0;
        while (j < 2)
        {
            int k := 0;
            while (k < 2)
            {
                sum := sum + c[i][j][k];
                k := k + 1;
            }
            j := j + 1;
        }
        i := i + 1;
    }
    PrintInt(sum);
}
