int isEven(int n)
{
    if (n = 0) { return 1; }
    if (n = 1) { return 0; }
    return isEven(n - 2);
}

int abs(int n)
{
    if (n < 0)
    {
        return 0 - n;
    }
    return n;
}

int max(int a, int b)
{
    if (a > b)
    {
        return a;
    }
    return b;
}

int min(int a, int b)
{
    if (a < b)
    {
        return a;
    }
    return b;
}

void main()
{
    PrintInt(isEven(0));
    PrintInt(isEven(1));
    PrintInt(isEven(4));
    PrintInt(isEven(7));
    PrintInt(abs(5));
    PrintInt(abs(0 - 5));
    PrintInt(abs(0));
    PrintInt(max(3, 7));
    PrintInt(max(10, 2));
    PrintInt(min(3, 7));
    PrintInt(min(10, 2));
}
