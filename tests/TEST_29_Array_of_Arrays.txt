array IntArray = int[];
array Matrix = IntArray[];

void main()
{
    Matrix mat := new IntArray[3];
    
    mat[0] := new int[3];
    mat[1] := new int[3];
    mat[2] := new int[3];
    
    mat[0][0] := 1;
    mat[0][1] := 2;
    mat[0][2] := 3;
    mat[1][0] := 4;
    mat[1][1] := 5;
    mat[1][2] := 6;
    mat[2][0] := 7;
    mat[2][1] := 8;
    mat[2][2] := 9;
    
    int i := 0;
    while (i < 3)
    {
        int j := 0;
        while (j < 3)
        {
            PrintInt(mat[i][j]);
            j := j + 1;
        }
        i := i + 1;
    }
}
