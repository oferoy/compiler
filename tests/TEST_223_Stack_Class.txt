array IntArr = int[];

class Stack
{
    IntArr data;
    int top := 0;
    int capacity := 0;

    void init(int cap)
    {
        data := new int[cap];
        capacity := cap;
        top := 0;
    }

    void push(int val)
    {
        if (top < capacity)
        {
            data[top] := val;
            top := top + 1;
        }
    }

    int pop()
    {
        if (top > 0)
        {
            top := top - 1;
            return data[top];
        }
        return 0 - 1;
    }

    int peek()
    {
        if (top > 0)
        {
            return data[top - 1];
        }
        return 0 - 1;
    }

    int isEmpty()
    {
        return top = 0;
    }
}

void main()
{
    Stack s := new Stack;
    s.init(10);
    s.push(10);
    s.push(20);
    s.push(30);
    PrintInt(s.peek());
    PrintInt(s.pop());
    PrintInt(s.pop());
    PrintInt(s.isEmpty());
    PrintInt(s.pop());
    PrintInt(s.isEmpty());
}
