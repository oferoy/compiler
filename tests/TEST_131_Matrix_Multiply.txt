array IntArr = int[];
array Matrix = IntArr[];

void main()
{
    Matrix a := new IntArr[2];
    a[0] := new int[2];
    a[1] := new int[2];
    a[0][0] := 1;
    a[0][1] := 2;
    a[1][0] := 3;
    a[1][1] := 4;

    Matrix b := new IntArr[2];
    b[0] := new int[2];
    b[1] := new int[2];
    b[0][0] := 5;
    b[0][1] := 6;
    b[1][0] := 7;
    b[1][1] := 8;

    Matrix c := new IntArr[2];
    c[0] := new int[2];
    c[1] := new int[2];
    int i := 0;
    while (i < 2)
    {
        int j := 0;
        while (j < 2)
        {
            int k := 0;
            int sum := 0;
            while (k < 2)
            {
                sum := sum + a[i][k] * b[k][j];
                k := k + 1;
            }
            c[i][j] := sum;
            j := j + 1;
        }
        i := i + 1;
    }
    PrintInt(c[0][0]);
    PrintInt(c[0][1]);
    PrintInt(c[1][0]);
    PrintInt(c[1][1]);
}
