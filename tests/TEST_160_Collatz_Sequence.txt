void main()
{
    int n := 27;
    int steps := 0;
    while (n = 1 = 0)
    {
        int isEven := n / 2 * 2 = n;
        if (isEven)
        {
            n := n / 2;
        }
        if (isEven = 0)
        {
            n := 3 * n + 1;
        }
        steps := steps + 1;
    }
    PrintInt(steps);
    PrintInt(n);
}
