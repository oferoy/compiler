class Counter {
    int count := 0;
    int getCount() { return count; }
    void inc() { count := count + 1; }
    void dec() { count := count - 1; }
    void addN(int n) { count := count + n; }
}

void main()
{
    Counter c := new Counter;
    c.inc();
    c.inc();
    c.inc();
    PrintInt(c.getCount());
    c.dec();
    PrintInt(c.getCount());
    c.addN(10);
    PrintInt(c.getCount());
    Counter d := c;
    d.inc();
    PrintInt(c.getCount());
    PrintInt(d.getCount());
    PrintInt(c = d);
}
