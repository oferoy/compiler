int fib(int n)
{
    if (n < 2)
    {
        return n;
    }
    
    int a := 0;
    int b := 1;
    int i := 2;
    int temp := 0;
    
    while (i < n + 1)
    {
        temp := a + b;
        a := b;
        b := temp;
        i := i + 1;
    }
    
    return b;
}

void main()
{
    PrintInt(fib(0));
    PrintInt(fib(1));
    PrintInt(fib(5));
    PrintInt(fib(10));
    PrintInt(fib(15));
}
