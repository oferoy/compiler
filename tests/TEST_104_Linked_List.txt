class Node {
    int val;
    Node next := nil;
}

void printList(Node n)
{
    while (n = nil = 0)
    {
        PrintInt(n.val);
        n := n.next;
    }
}

void main()
{
    Node a := new Node;
    Node b := new Node;
    Node c := new Node;
    a.val := 1;
    b.val := 2;
    c.val := 3;
    a.next := b;
    b.next := c;
    printList(a);
}
